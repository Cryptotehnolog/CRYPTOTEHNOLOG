#### 0.5.2. Сохранение в Vault

```powershell
code scripts\vault_setup.py
```

```python
"""
Vault Setup Script
Сохранение секретов в HashiCorp Vault
"""

import hvac
import sys

# Настройки
VAULT_ADDR = "http://localhost:8200"
VAULT_TOKEN = "dev-only-token"

def main():
    # Подключение к Vault
    client = hvac.Client(url=VAULT_ADDR, token=VAULT_TOKEN)

    if not client.is_authenticated():
        print("❌ Vault authentication failed!")
        sys.exit(1)

    print("✅ Connected to Vault")

    # Bybit secrets
    print("\n📝 Setting Bybit secrets...")
    client.secrets.kv.v2.create_or_update_secret(
```

```python
"""
Complete Environment Test
Проверка всех компонентов Phase 0
"""

import sys
import asyncio
import redis
import asyncpg
import hvac
from decimal import Decimal


async def test_all():
    print("🧪 CRYPTOTEHNOLOG Environment Test\n")

    errors = []

    # 1. Python version
    print("1️⃣ Python version...")
    if sys.version_info >= (3, 11):
        print(f"   ✅ Python {sys.version_info.major}.{sys.version_info.minor}")
    else:
        errors.append("Python version < 3.11")
        print("   ❌ Python version too old")

    # 2. Redis
    print("\n2️⃣ Redis connection...")
    try:
        r = redis.Redis(host='localhost', port=6379, socket_connect_timeout=5)
        r.ping()
        print("   ✅ Redis OK")
    except Exception as e:
        errors.append(f"Redis: {e}")
        print(f"   ❌ Redis failed: {e}")

    # 3. PostgreSQL
    print("\n3️⃣ PostgreSQL/TimescaleDB connection...")
    try:
        conn = await asyncpg.connect(
            host='localhost',
            port=5432,
            user='bot_user',
            password='bot_password_dev',
            database='trading_dev',
            timeout=10
        )
        version = await conn.fetchval('SELECT version()')
        print(f"   ✅ PostgreSQL OK: {version[:50]}...")
        await conn.close()
    except Exception as e:
        errors.append(f"PostgreSQL: {e}")
        print(f"   ❌ PostgreSQL failed: {e}")

    # 4. Vault
    print("\n4️⃣ HashiCorp Vault...")
    try:
        client = hvac.Client(url='http://localhost:8200', token='dev-only-token')
        if client.is_authenticated():
            print("   ✅ Vault authenticated")
        else:
            errors.append("Vault authentication failed")
            print("   ❌ Vault not authenticated")
    except Exception as e:
        errors.append(f"Vault: {e}")
        print(f"   ❌ Vault failed: {e}")

    # 5. Pydantic
    print("\n5️⃣ Pydantic models...")
    try:
        from pydantic import BaseModel, Field

        class TestModel(BaseModel):
            value: Decimal = Field(gt=0)

        model = TestModel(value=Decimal("1.5"))
        assert model.value == Decimal("1.5")
        print("   ✅ Pydantic OK")
    except Exception as e:
        errors.append(f"Pydantic: {e}")
        print(f"   ❌ Pydantic failed: {e}")

    # 6. CCXT
    print("\n6️⃣ CCXT library...")
    try:
        import ccxt
        print(f"   ✅ CCXT {ccxt.__version__} OK")
    except Exception as e:
        errors.append(f"CCXT: {e}")
        print(f"   ❌ CCXT failed: {e}")

    # Summary
    print("\n" + "="*50)
    if not errors:
        print("✅✅✅ ALL TESTS PASSED! Environment ready! ✅✅✅")
        print("\nNext step: Phase 1 - Infrastructure Core")
        return True
    else:
        print("❌ ERRORS DETECTED:")
        for err in errors:
            print(f"  - {err}")
        print("\nFix errors before proceeding to Phase 1!")
        return False


if __name__ == "__main__":
    result = asyncio.run(test_all())
    sys.exit(0 if result else 1)
```

## 🐛 TROUBLESHOOTING

### Python не найден в PATH

**Проблема:** `python` не распознается в PowerShell

**Решение:**

1. Переустановите Python с галочкой "Add to PATH"
2. Или добавьте вручную в System Environment Variables:
   ```
   C:\Users\ВашИмя\AppData\Local\Programs\Python\Python311
   C:\Users\ВашИмя\AppData\Local\Programs\Python\Python311\Scripts
   ```
3. Перезапустите PowerShell

---

### Docker контейнеры не запускаются

**Проблема:** `docker-compose up` падает с ошибками

**Решение:**

1. Убедитесь, что Docker Desktop запущен (зеленый индикатор)
2. Проверьте логи: `docker-compose logs`
3. Попробуйте пересоздать: `docker-compose down -v && docker-compose up -d`

---

### Vault недоступен

**Проблема:** Ошибка подключения к Vault

**Решение:**

1. Проверьте статус: `docker-compose ps vault`
2. Если контейнер "unhealthy", перезапустите: `docker-compose restart vault`
3. Проверьте логи: `docker-compose logs vault`

---

### PowerShell Execution Policy ошибка

**Проблема:** `.\venv\Scripts\Activate.ps1` не работает

**Решение:**

```powershell
Set-ExecutionPolicy -ExecutionPolicy RemoteSigned -Scope CurrentUser
```
