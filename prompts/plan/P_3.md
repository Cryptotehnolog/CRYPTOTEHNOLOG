/// Приоритеты событий
#[derive(Debug, Clone, Copy, PartialEq, Eq, Ordering, Serialize, Deserialize)]
pub enum Priority {
    Critical = 0,  // Немедленная обработка, никогда не дропать
    High = 1,      // Важные события, дропать только при extreme pressure
    Normal = 2,    // Обычные события
    Low = 3,       // Фоновые события, дропать первыми
}

impl Default for Priority {
    fn default() -> Self { Priority::Normal }
}

// Методы
impl Priority {
    pub fn queue_capacity(&self) -> usize;
    pub fn requires_persistence(&self) -> bool;
    pub fn is_droppable(&self) -> bool;
    pub fn as_u8(&self) -> u8;
}
```

/// Стратегия backpressure при переполнении
pub enum BackpressureStrategy {
    DropLow,        // Дропать low priority
    OverflowNormal, // Переполнение в normal очередь
    DropNormal,     // Дропать normal + low
    BlockCritical,  // Блокировать только critical
}

/// Обработчик backpressure
pub struct BackpressureHandler {
    strategy: BackpressureStrategy,
    dropped_low: Arc<AtomicU64>,
    dropped_normal: Arc<AtomicU64>,
    dropped_high: Arc<AtomicU64>,
    blocked_critical: Arc<AtomicU64>,
}

impl BackpressureHandler {
    pub fn new(strategy: BackpressureStrategy) -> Self;
    pub fn handle(&self, event: &Event, queue: &mut PriorityQueue) -> HandleResult;
    pub fn get_dropped_stats(&self) -> DroppedStats;
}
```
